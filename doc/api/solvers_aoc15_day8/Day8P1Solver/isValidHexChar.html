bool isValidHexChar(String char) {
  int codeUnit = char.codeUnitAt(0);
  return (codeUnit >= '0'.codeUnitAt(0) && codeUnit <= '9'.codeUnitAt(0)) ||
      (codeUnit >= 'a'.codeUnitAt(0) && codeUnit <= 'f'.codeUnitAt(0)) ||
      (codeUnit >= 'A'.codeUnitAt(0) && codeUnit <= 'F'.codeUnitAt(0));
}
